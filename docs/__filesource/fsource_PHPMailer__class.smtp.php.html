  99.    * Connect to the server specified on the port specified.
  100.    * If the port is not specified use the default SMTP_PORT.
  101.    * If tval is specified then a connection will try and be
  102.    * established with the server for that number of seconds.
  103.    * If tval is not specified the default is 30 seconds to
  104.    * try on the connection.
  105.    *
  106.    * SMTP CODE SUCCESS: 220
  107.    * SMTP CODE FAILURE: 421
  108.    * @access public
  109.    * @return bool 
  110.    */
  111.   function Connect($host,$port=0,$tval=30) {
  112.     /* set the error val to null so there is no confusion */
  113.     $this->error = null;
  114.  
  115.     /* make sure we are __not__ connected */
  116.     if($this->connected()) {
  117.       /* ok we are connected! what should we do?
  118.        * for now we will just give an error saying we
  119.        * are already connected
  120.        */
  121.       $this->error = array("error" => "Already connected to a server");
  122.       return false;
  123.     }
  124.  
  125.     if(empty($port)) {
  126.       $port = $this->SMTP_PORT;
  127.     }
  128.  
  129.     /* connect to the smtp server */
  130.     $this->smtp_conn = fsockopen($host,    // the host of the server
  131.                                  $port,    // the port to use
  132.                                  $errno,   // error number if any
  133.                                  $errstr,  // error message if any
  134.                                  $tval);   // give up after ? secs
  135.     /* verify we connected properly */
  136.     if(empty($this->smtp_conn)) {
  137.       $this->error = array("error" => "Failed to connect to server",
  138.                            "errno" => $errno,
  139.                            "errstr" => $errstr);
  140.       if($this->do_debug >= 1) {
  141.         echo "SMTP -> ERROR: " . $this->error["error"] .
  142.                  ": $errstr ($errno)" . $this->CRLF;
  143.       }
  144.       return false;
  145.     }
  146.  
  147.     /* sometimes the SMTP server takes a little longer to respond
  148.      * so we will give it a longer timeout for the first read
  149.      * - Windows still does not have support for this timeout function
  150.      */
  151.     if(substr(PHP_OS, 0, 3) != "WIN")
  152.      socket_set_timeout($this->smtp_conn, $tval, 0);
  153.  
  154.     /* get any announcement stuff */
  155.     $announce = $this->get_lines();
  156.  
  157.     /* set the timeout  of any socket functions at 1/10 of a second */
  158.     //if(function_exists("socket_set_timeout"))
  159.     //   socket_set_timeout($this->smtp_conn, 0, 100000);
  160.  
  161.     if($this->do_debug >= 2) {
  162.       echo "SMTP -> FROM SERVER:" . $this->CRLF . $announce;
  163.     }
  164.  
  165.     return true;
  166.   }

  168.   /**
  169.    * Performs SMTP authentication.  Must be run after running the

  174.   function Authenticate($username, $password) {
  175.     // Start authentication
  176.     fputs($this->smtp_conn,"AUTH LOGIN" . $this->CRLF);
  177.  
  178.     $rply = $this->get_lines();
  179.     $code = substr($rply,0,3);
  180.  
  181.     if($code != 334) {
  182.       $this->error =
  183.         array("error" => "AUTH not accepted from server",
  184.               "smtp_code" => $code,
  185.               "smtp_msg" => substr($rply,4));
  186.       if($this->do_debug >= 1) {
  187.         echo "SMTP -> ERROR: " . $this->error["error"] .
  188.                  ": " . $rply . $this->CRLF;
  189.       }
  190.       return false;
  191.     }
  192.  
  193.     // Send encoded username
  194.     fputs($this->smtp_conn, base64_encode($username) . $this->CRLF);
  195.  
  196.     $rply = $this->get_lines();

  198.  
  199.     if($code != 334) {
  200.       $this->error =
  201.         array("error" => "Username not accepted from server",
  202.               "smtp_code" => $code,
  203.               "smtp_msg" => substr($rply,4));
  204.       if($this->do_debug >= 1) {
  205.         echo "SMTP -> ERROR: " . $this->error["error"] .
  206.                  ": " . $rply . $this->CRLF;
  207.       }
  208.       return false;
  209.     }
  210.  
  211.     // Send encoded password

  214.     $rply = $this->get_lines();
  215.     $code = substr($rply,0,3);
  216.

  221.               "smtp_msg" => substr($rply,4));
  222.       if($this->do_debug >= 1) {
  223.         echo "SMTP -> ERROR: " . $this->error["error"] .
  224.                  ": " . $rply . $this->CRLF;
  225.       }
  226.       return false;
  227.     }
  228.  
  229.     return true;
  230.   }

  276.   /**
  277.    * Issues a data command and sends the msg_data to the server
  278.    * finializing the mail transaction. $msg_data is the message
  279.    * that is to be send with the headers. Each header needs to be
  280.    * on a single line followed by a <CRLF> with the message headers
  281.    * and the message body being seperated by and additional <CRLF>.
  282.    *
  283.    * Implements rfc 821: DATA <CRLF>
  284.    *
  285.    * SMTP CODE INTERMEDIATE: 354
  286.    *     [data]
  287.    *     <CRLF>.<CRLF>
  288.    *     SMTP CODE SUCCESS: 250
  289.    *     SMTP CODE FAILURE: 552,554,451,452
  290.    * SMTP CODE FAILURE: 451,554
  291.    * SMTP CODE ERROR  : 500,501,503,421
  292.    * @access public
  293.    * @return bool 
  294.    */
  295.   function Data($msg_data) {
  296.     $this->error = null; // so no confusion is caused
  297.  
  298.     if(!$this->connected()) {
  299.       $this->error = array(
  300.               "error" => "Called Data() without being connected");
  301.       return false;
  302.     }
  303.  
  304.     fputs($this->smtp_conn,"DATA" . $this->CRLF);
  305.  
  306.     $rply = $this->get_lines();
  307.     $code = substr($rply,0,3);
  308.  
  309.     if($this->do_debug >= 2) {
  310.       echo "SMTP -> FROM SERVER:" . $this->CRLF . $rply;
  311.     }
  312.  
  313.     if($code != 354) {
  314.       $this->error =
  315.         array("error" => "DATA command not accepted from server",
  316.               "smtp_code" => $code,
  317.               "smtp_msg" => substr($rply,4));
  318.       if($this->do_debug >= 1) {
  319.         echo "SMTP -> ERROR: " . $this->error["error"] .
  320.                  ": " . $rply . $this->CRLF;
  321.       }
  322.       return false;
  323.     }
  324.  
  325.     /* the server is ready to accept data!
  326.      * according to rfc 821 we should not send more than 1000
  327.      * including the CRLF
  328.      * characters on a single line so we will break the data up
  329.      * into lines by \r and/or \n then if needed we will break
  330.      * each of those into smaller lines to fit within the limit.
  331.      * in addition we will be looking for lines that start with
  332.      * a period '.' and append and additional period '.' to that
  333.      * line. NOTE: this does not count towards are limit.
  334.      */
  335.  
  336.     // normalize the line breaks so we know the explode works
  337.     $msg_data = str_replace("\r\n","\n",$msg_data);
  338.     $msg_data = str_replace("\r","\n",$msg_data);
  339.     $lines = explode("\n",$msg_data);
  340.  
  341.     /* we need to find a good way to determine is headers are
  342.      * in the msg_data or if it is a straight msg body
  343.      * currently I am assuming rfc 822 definitions of msg headers
  344.      * and if the first field of the first line (':' sperated)
  345.      * does not contain a space then it _should_ be a header
  346.      * and we can process all lines before a blank "" line as
  347.      * headers.
  348.      */
  349.     $field = substr($lines[0],0,strpos($lines[0],":"));
  350.     $in_headers = false;
  351.     if(!empty($field) && !strstr($field," ")) {
  352.       $in_headers = true;
  353.     }
  354.  
  355.     $max_line_length = 998; // used below; set here for ease in change
  356.  
  357.     while(list(,$line) = @each($lines)) {
  358.       $lines_out = null;
  359.       if($line == "" && $in_headers) {
  360.         $in_headers = false;
  361.       }
  362.       // ok we need to break this line up into several smaller lines
  363.       while(strlen($line) > $max_line_length) {
  364.         $pos = strrpos(substr($line,0,$max_line_length)," ");
  365.  
  366.         // Patch to fix DOS attack
  367.         if(!$pos) {
  368.           $pos = $max_line_length - 1;
  369.         }
  370.  
  371.         $lines_out[] = substr($line,0,$pos);
  372.         $line = substr($line,$pos + 1);
  373.         /* if we are processing headers we need to
  374.          * add a LWSP-char to the front of the new line
  375.          * rfc 822 on long msg headers
  376.          */
  377.         if($in_headers) {
  378.           $line = "\t" . $line;
  379.         }
  380.       }
  381.       $lines_out[] = $line;
  382.  
  383.       // now send the lines to the server
  384.       while(list(,$line_out) = @each($lines_out)) {
  385.         if(strlen($line_out) > 0)
  386.         {
  387.           if(substr($line_out, 0, 1) == ".") {
  388.             $line_out = "." . $line_out;
  389.           }
  390.         }
  391.         fputs($this->smtp_conn,$line_out . $this->CRLF);
  392.       }
  393.     }
  394.  
  395.     // ok all the message data has been sent so lets get this
  396.     // over with aleady
  397.     fputs($this->smtp_conn, $this->CRLF . "." . $this->CRLF);
  398.  
  399.     $rply = $this->get_lines();
  400.     $code = substr($rply,0,3);
  401.  
  402.     if($this->do_debug >= 2) {
  403.       echo "SMTP -> FROM SERVER:" . $this->CRLF . $rply;
  404.     }
  405.  
  406.     if($code != 250) {
  407.       $this->error =
  408.         array("error" => "DATA not accepted from server",
  409.               "smtp_code" => $code,
  410.               "smtp_msg" => substr($rply,4));
  411.       if($this->do_debug >= 1) {
  412.         echo "SMTP -> ERROR: " . $this->error["error"] .
  413.                  ": " . $rply . $this->CRLF;
  414.       }
  415.       return false;
  416.     }
  417.     return true;
  418.   }
  419.  
  420.   /**
  421.    * Expand takes the name and asks the server to list all the
  422.    * people who are members of the _list_. Expand will return
  423.    * back and array of the result or false if an error occurs.
  424.    * Each value in the array returned has the format of:
  425.    *     [ <full-name> <sp> ] <path>
  426.    * The definition of <path> is defined in rfc 821
  427.    *
