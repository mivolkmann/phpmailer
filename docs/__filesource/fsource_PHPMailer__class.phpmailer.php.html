  376.    * not sent successfully then it returns false.  Use the ErrorInfo

  380.   function Send() {
  381.     $header = '';
  382.     $body = '';
  383.     $result = true;
  384.  
  385.     if((count($this->to) + count($this->cc) + count($this->bcc)) < 1) {
  386.       $this->SetError($this->Lang('provide_address'));
  387.       return false;
  388.     }
  389.  
  390.     /* Set whether the message is multipart/alternative */

  393.     }
  394.

  397.     $header .= $this->CreateHeader();
  398.     $body = $this->CreateBody();
  399.  
  400.     if($body == '') {
  401.       return false;
  402.     }
  403.

  405.     switch($this->Mailer) {
  406.       case 'sendmail':
  407.         $result = $this->SendmailSend($header, $body);

  414.         break;
  415.       default:
  416.         $result = $this->MailSend($header, $body);
  417.         break;
  418.         //$this->SetError($this->Mailer . $this->Lang('mailer_not_supported'));
  419.         //$result = false;
  420.         //break;
  421.     }
  422.  
  423.     return $result;
  424.   }

  431.   function SendmailSend($header, $body) {
  432.     if ($this->Sender != '') {
  433.       $sendmail = sprintf("%s -oi -f %s -t", escapeshellcmd($this->Sendmail), escapeshellarg($this->Sender));
  434.     } else {
  435.       $sendmail = sprintf("%s -oi -t", escapeshellcmd($this->Sendmail));
  436.     }
  437.  
  438.     if(!@$mail = popen($sendmail, 'w')) {
  439.       $this->SetError($this->Lang('execute') . $this->Sendmail);
  440.       return false;
  441.     }
  442.  
  443.     fputs($mail, $header);
  444.     fputs($mail, $body);
  445.  
  446.     $result = pclose($mail) >> 8 & 0xFF;
  447.     if($result != 0) {
  448.       $this->SetError($this->Lang('execute') . $this->Sendmail);
  449.       return false;
  450.     }
  451.  
  452.     return true;
  453.   }

  455.   /**
  456.    * Sends mail using the PHP mail() function.
  457.    * @access private
  458.    * @return bool 
  459.    */
  460.   function MailSend($header, $body) {
  461.  
  462.     $to = '';
  463.     for($i = 0; $i < count($this->to); $i++) {
  464.       if($i != 0) { $to .= ', '; }
  465.       $to .= $this->AddrFormat($this->to[$i]);
  466.     }
  467.  
  468.     $toArr = split(',', $to);
  469.  
  470.     if ($this->Sender != '' && strlen(ini_get('safe_mode'))< 1) {
  471.       $old_from = ini_get('sendmail_from');
  472.       ini_set('sendmail_from', $this->Sender);
  473.       $params = sprintf("-oi -f %s", $this->Sender);
  474.       if ($this->SingleTo === true && count($toArr) > 1) {
  475.         foreach ($toArr as $key => $val) {
  476.           $rt = @mail($val, $this->EncodeHeader($this->SecureHeader($this->Subject)), $body, $header, $params);
  477.         }
  478.       } else {
  479.         $rt = @mail($to, $this->EncodeHeader($this->SecureHeader($this->Subject)), $body, $header, $params);
  480.       }
  481.     } else {
  482.       if ($this->SingleTo === true && count($toArr) > 1) {
  483.         foreach ($toArr as $key => $val) {
  484.           $rt = @mail($val, $this->EncodeHeader($this->SecureHeader($this->Subject)), $body, $header, $params);
  485.         }
  486.       } else {
  487.         $rt = @mail($to, $this->EncodeHeader($this->SecureHeader($this->Subject)), $body, $header);
  488.       }
  489.     }
  490.  
  491.     if (isset($old_from)) {
  492.       ini_set('sendmail_from', $old_from);
  493.     }
  494.  
  495.     if(!$rt) {
  496.       $this->SetError($this->Lang('instantiate'));
  497.       return false;
  498.     }
  499.

  503.   /**
  504.    * Sends mail via SMTP using PhpSMTP
  505.    * (Author: Chris Ryan).
  506.    * Returns false if there is a bad MAIL FROM, RCPT, or DATA input.
  507.    * @uses SMTP
  508.    * @access private
  509.    * @return bool 
  510.    */
  511.   function SmtpSend($header, $body) {
  512.     include_once($this->PluginDir . 'class.smtp.php');
  513.     $error = '';
  514.     $bad_rcpt = array();
  515.  
  516.     if(!$this->SmtpConnect()) {
  517.       return false;
  518.     }
  519.  
  520.     $smtp_from = ($this->Sender == '') ? $this->From : $this->Sender;
  521.     if(!$this->smtp->Mail($smtp_from)) {
  522.       $error = $this->Lang('from_failed') . $smtp_from;
  523.       $this->SetError($error);
  524.       $this->smtp->Reset();
  525.       return false;
  526.     }
  527.  
  528.     /* Attempt to send attach all recipients */
  529.     for($i = 0; $i < count($this->to); $i++) {
  530.       if(!$this->smtp->Recipient($this->to[$i][0])) {
  531.         $bad_rcpt[] = $this->to[$i][0];
  532.       }
  533.     }
  534.     for($i = 0; $i < count($this->cc); $i++) {
  535.       if(!$this->smtp->Recipient($this->cc[$i][0])) {
  536.         $bad_rcpt[] = $this->cc[$i][0];
  537.       }
  538.     }
  539.     for($i = 0; $i < count($this->bcc); $i++) {
  540.       if(!$this->smtp->Recipient($this->bcc[$i][0])) {
  541.         $bad_rcpt[] = $this->bcc[$i][0];
  542.       }
  543.     }
  544.  
  545.     if(count($bad_rcpt) > 0) { // Create error message
  546.       for($i = 0; $i < count($bad_rcpt); $i++) {
  547.         if($i != 0) {
  548.           $error .= ', ';
  549.         }
  550.         $error .= $bad_rcpt[$i];
  551.       }
  552.       $error = $this->Lang('recipients_failed') . $error;
  553.       $this->SetError($error);
  554.       $this->smtp->Reset();
  555.       return false;
  556.     }
  557.  
  558.     if(!$this->smtp->Data($header . $body)) {
  559.       $this->SetError($this->Lang('data_not_accepted'));
  560.       $this->smtp->Reset();
  561.       return false;
  562.     }
  563.     if($this->SMTPKeepAlive == true) {
  564.       $this->smtp->Reset();
  565.     } else {
  566.       $this->SmtpClose();
  567.     }
  568.  
  569.     return true;
  570.   }

  572.   /**
  573.    * Initiates a connection to an SMTP server.
  574.    * Returns false if the operation failed.
  575.    * @uses SMTP
  576.    * @access private
  577.    * @return bool 
  578.    */
  579.   function SmtpConnect() {
  580.     if($this->smtp == NULL) {
  581.       $this->smtp = new SMTP();
  582.     }
  583.  
  584.     $this->smtp->do_debug = $this->SMTPDebug;
  585.     $hosts = explode(';', $this->Host);
  586.     $index = 0;
  587.     $connection = ($this->smtp->Connected());
  588.  
  589.     /* Retry while there is no connection */
  590.     while($index < count($hosts) && $connection == false) {
  591.       $hostinfo = array();
  592.       if(eregi('^(.+):([0-9]+)$', $hosts[$index], $hostinfo)) {
  593.         $host = $hostinfo[1];
  594.         $port = $hostinfo[2];
  595.       } else {
  596.         $host = $hosts[$index];
  597.         $port = $this->Port;
  598.       }
  599.  
  600.       if($this->smtp->Connect(((!empty($this->SMTPSecure))?$this->SMTPSecure.'://':'').$host, $port, $this->Timeout)) {
  601.         if ($this->Helo != '') {
  602.           $this->smtp->Hello($this->Helo);
  603.         } else {
  604.           $this->smtp->Hello($this->ServerHostname());
  605.         }
  606.  
  607.         $connection = true;
  608.         if($this->SMTPAuth) {
  609.           if(!$this->smtp->Authenticate($this->Username, $this->Password)) {
  610.             $this->SetError($this->Lang('authenticate'));
  611.             $this->smtp->Reset();
  612.             $connection = false;
  613.           }
  614.         }
  615.       }
  616.       $index++;
  617.     }
  618.     if(!$connection) {
  619.       $this->SetError($this->Lang('connect_host'));
  620.     }
  621.  
  622.     return $connection;
  623.   }

  699.   /**
  700.    * Wraps message for use with mailers that do not
  701.    * automatically perform wrapping and for quoted-printable.
  702.    * Original written by philippe.
  703.    * @access private
  704.    * @return string 
  705.    */
  706.   function WrapText($message, $length, $qp_mode = false) {
  707.     $soft_break = ($qp_mode) ? sprintf(" =%s", $this->LE) : $this->LE;
  708.  
  709.     $message = $this->FixEOL($message);
  710.     if (substr($message, -1) == $this->LE) {
  711.       $message = substr($message, 0, -1);
  712.     }
  713.  
  714.     $line = explode($this->LE, $message);
  715.     $message = '';
  716.     for ($i=0 ;$i < count($line); $i++) {
  717.       $line_part = explode(' ', $line[$i]);
  718.       $buf = '';
  719.       for ($e = 0; $e<count($line_part); $e++) {
  720.         $word = $line_part[$e];
  721.         if ($qp_mode and (strlen($word) > $length)) {
  722.           $space_left = $length - strlen($buf) - 1;
  723.           if ($e != 0) {
  724.             if ($space_left > 20) {
  725.               $len = $space_left;
  726.               if (substr($word, $len - 1, 1) == '=') {
  727.                 $len--;
  728.               } elseif (substr($word, $len - 2, 1) == '=') {
  729.                 $len -= 2;
  730.               }
  731.               $part = substr($word, 0, $len);
  732.               $word = substr($word, $len);
  733.               $buf .= ' ' . $part;
  734.               $message .= $buf . sprintf("=%s", $this->LE);
  735.             } else {
  736.               $message .= $buf . $soft_break;
  737.             }
  738.             $buf = '';
  739.           }
  740.           while (strlen($word) > 0) {
  741.             $len = $length;
  742.             if (substr($word, $len - 1, 1) == '=') {
  743.               $len--;
  744.             } elseif (substr($word, $len - 2, 1) == '=') {
  745.               $len -= 2;
  746.             }
  747.             $part = substr($word, 0, $len);
  748.             $word = substr($word, $len);
  749.  
  750.             if (strlen($word) > 0) {
  751.               $message .= $part . sprintf("=%s", $this->LE);
  752.             } else {
  753.               $buf = $part;
  754.             }
  755.           }
  756.         } else {
  757.           $buf_o = $buf;
  758.           $buf .= ($e == 0) ? $word : (' ' . $word);
  759.  
  760.           if (strlen($buf) > $length and $buf_o != '') {
  761.             $message .= $buf_o . $soft_break;
  762.             $buf = $word;
  763.           }
  764.         }
  765.       }
  766.       $message .= $buf . $this->LE;
  767.     }
  768.  
  769.     return $message;
  770.   }

  781.  
  782.     switch($this->message_type) {

  794.   /**
  795.    * Assembles message header.
  796.    * @access private
  797.    * @return string 
  798.    */
  799.   function CreateHeader() {
  800.     $result = '';
  801.  
  802.     /* Set the boundaries */
  803.     $uniq_id = md5(uniqid(time()));
  804.     $this->boundary[1] = 'b1_' . $uniq_id;
  805.     $this->boundary[2] = 'b2_' . $uniq_id;
  806.  
  807.     $result .= $this->HeaderLine('Date', $this->RFCDate());
  808.     if($this->Sender == '') {
  809.       $result .= $this->HeaderLine('Return-Path', trim($this->From));
  810.     } else {
  811.       $result .= $this->HeaderLine('Return-Path', trim($this->Sender));
  812.     }
  813.  
  814.     /* To be created automatically by mail() */
  815.     if($this->Mailer != 'mail') {
  816.       if(count($this->to) > 0) {
  817.         $result .= $this->AddrAppend('To', $this->to);
  818.       } elseif (count($this->cc) == 0) {
  819.         $result .= $this->HeaderLine('To', 'undisclosed-recipients:;');
  820.       }
  821.       if(count($this->cc) > 0) {
  822.         $result .= $this->AddrAppend('Cc', $this->cc);
  823.       }
  824.     }
  825.  
  826.     $from = array();
  827.     $from[0][0] = trim($this->From);
  828.     $from[0][1] = $this->FromName;
  829.     $result .= $this->AddrAppend('From', $from);
  830.  
  831.     /* sendmail and mail() extract Cc from the header before sending */
  832.     if((($this->Mailer == 'sendmail') || ($this->Mailer == 'mail')) && (count($this->cc) > 0)) {
  833.       $result .= $this->AddrAppend('Cc', $this->cc);
  834.     }
  835.  
  836.     /* sendmail and mail() extract Bcc from the header before sending */
  837.     if((($this->Mailer == 'sendmail') || ($this->Mailer == 'mail')) && (count($this->bcc) > 0)) {
  838.       $result .= $this->AddrAppend('Bcc', $this->bcc);
  839.     }
  840.  
  841.     if(count($this->ReplyTo) > 0) {
  842.       $result .= $this->AddrAppend('Reply-to', $this->ReplyTo);
  843.     }
  844.  
  845.     /* mail() sets the subject itself */
  846.     if($this->Mailer != 'mail') {
  847.       $result .= $this->HeaderLine('Subject', $this->EncodeHeader($this->SecureHeader($this->Subject)));
  848.     }
  849.  
  850.     $result .= sprintf("Message-ID: <%s@%s>%s", $uniq_id, $this->ServerHostname(), $this->LE);
  851.     $result .= $this->HeaderLine('X-Priority', $this->Priority);
  852.     $result .= $this->HeaderLine('X-Mailer', 'PHPMailer (phpmailer.sourceforge.net) [version ' . $this->Version . ']');
  853.  
  854.     if($this->ConfirmReadingTo != '') {
  855.       $result .= $this->HeaderLine('Disposition-Notification-To', '<' . trim($this->ConfirmReadingTo) . '>');
  856.     }
  857.  
  858.     // Add custom headers
  859.     for($index = 0; $index < count($this->CustomHeader); $index++) {
  860.       $result .= $this->HeaderLine(trim($this->CustomHeader[$index][0]), $this->EncodeHeader(trim($this->CustomHeader[$index][1])));
  861.     }
  862.     $result .= $this->HeaderLine('MIME-Version', '1.0');
  863.  
  864.     switch($this->message_type) {
  865.       case 'plain':
  866.         $result .= $this->HeaderLine('Content-Transfer-Encoding', $this->Encoding);
  867.         $result .= sprintf("Content-Type: %s; charset=\"%s\"", $this->ContentType, $this->CharSet);
  868.         break;
  869.       case 'attachments':
  870.         /* fall through */
  871.       case 'alt_attachments':
  872.         if($this->InlineImageExists()){
  873.           $result .= sprintf("Content-Type: %s;%s\ttype=\"text/html\";%s\tboundary=\"%s\"%s", 'multipart/related', $this->LE, $this->LE, $this->boundary[1], $this->LE);
  874.         } else {
  875.           $result .= $this->HeaderLine('Content-Type', 'multipart/mixed;');
  876.           $result .= $this->TextLine("\tboundary=\"" . $this->boundary[1] . '"');
  877.         }
  878.         break;
  879.       case 'alt':
  880.         $result .= $this->HeaderLine('Content-Type', 'multipart/alternative;');
  881.         $result .= $this->TextLine("\tboundary=\"" . $this->boundary[1] . '"');
  882.         break;
  883.     }
  884.  
  885.     if($this->Mailer != 'mail') {
  886.       $result .= $this->LE.$this->LE;
  887.     }
  888.  
  889.     return $result;
  890.   }

  892.   /**
  893.    * Assembles the message body.  Returns an empty string on failure.
  894.    * @access private
  895.    * @return string 
  896.    */
  897.   function CreateBody() {
  898.     $result = '';
  899.  
  900.     $this->SetWordWrap();
  901.  
  902.     switch($this->message_type) {
  903.       case 'alt':
  904.         $result .= $this->GetBoundary($this->boundary[1], '', 'text/plain', '');
  905.         $result .= $this->EncodeString($this->AltBody, $this->Encoding);
  906.         $result .= $this->LE.$this->LE;
  907.         $result .= $this->GetBoundary($this->boundary[1], '', 'text/html', '');
  908.         $result .= $this->EncodeString($this->Body, $this->Encoding);
  909.         $result .= $this->LE.$this->LE;
  910.         $result .= $this->EndBoundary($this->boundary[1]);
  911.         break;
  912.       case 'plain':
  913.         $result .= $this->EncodeString($this->Body, $this->Encoding);
  914.         break;
  915.       case 'attachments':
  916.         $result .= $this->GetBoundary($this->boundary[1], '', '', '');
  917.         $result .= $this->EncodeString($this->Body, $this->Encoding);
  918.         $result .= $this->LE;
  919.         $result .= $this->AttachAll();
  920.         break;
  921.       case 'alt_attachments':
  922.         $result .= sprintf("--%s%s", $this->boundary[1], $this->LE);
  923.         $result .= sprintf("Content-Type: %s;%s" . "\tboundary=\"%s\"%s", 'multipart/alternative', $this->LE, $this->boundary[2], $this->LE.$this->LE);
  924.         $result .= $this->GetBoundary($this->boundary[2], '', 'text/plain', '') . $this->LE; // Create text body
  925.         $result .= $this->EncodeString($this->AltBody, $this->Encoding);
