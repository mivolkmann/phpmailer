  137.   /**
  138.    * Combination of public events - connect, login, disconnect
  139.    *
  140.    * @param string $host 
  141.    * @param integer $port 
  142.    * @param integer $tval 
  143.    * @param string $username 
  144.    * @param string $password

  146.   function Authorise ($host, $port = false, $tval = false, $username, $password, $debug_level = 0) {
  147.     $this->host = $host;
  148.  
  149.     //  If no port value is passed, retrieve it
  150.     if ($port == false) {
  151.       $this->port = $this->POP3_PORT;
  152.     } else {
  153.       $this->port = $port;
  154.     }
  155.  
  156.     //  If no port value is passed, retrieve it
  157.     if ($tval == false) {
  158.       $this->tval = $this->POP3_TIMEOUT;
  159.     } else {
  160.       $this->tval = $tval;
  161.     }
  162.  
  163.     $this->do_debug = $debug_level;
  164.     $this->username = $username;
  165.     $this->password = $password;
  166.  
  167.     //  Refresh the error log
  168.     $this->error = null;
  169.  
  170.     //  Connect
  171.     $result = $this->Connect($this->host, $this->port, $this->tval);
  172.  
  173.     if ($result) {
  174.       $login_result = $this->Login($this->username, $this->password);
  175.  
  176.       if ($login_result) {
  177.         $this->Disconnect();
  178.  
  179.         return true;
  180.       }
  181.  
  182.     }
  183.  
  184.     //  We need to disconnect regardless if the login succeeded
  185.     $this->Disconnect();
  186.  
  187.     return false;
  188.   }

  190.   /**
  191.    * Connect to the POP3 server
  192.    *
  193.    * @param string $host 
  194.    * @param integer $port 
  195.    * @param integer $tval 
  196.    * @return boolean 
  197.    */
  198.   function Connect ($host, $port = false, $tval = 30) {
  199.     //  Are we already connected?
  200.     if ($this->connected) {
  201.       return true;
  202.     }
  203.  
  204.     /*
  205.     On Windows this will raise a PHP Warning error if the hostname doesn't exist.
  206.     Rather than supress it with @fsockopen, let's capture it cleanly instead
  207.     */
  208.  
  209.     set_error_handler(array(&$this, 'catchWarning'));
  210.  
  211.     //  Connect to the POP3 server
  212.     $this->pop_conn = fsockopen($host,    //  POP3 Host
  213.                   $port,    //  Port #
  214.                   $errno,   //  Error Number
  215.                   $errstr,  //  Error Message
  216.                   $tval);   //  Timeout (seconds)
  217.  
  218.     //  Restore the error handler
  219.     restore_error_handler();
  220.  
  221.     //  Does the Error Log now contain anything?
  222.     if ($this->error && $this->do_debug >= 1) {
  223.       $this->displayErrors();
  224.     }
  225.  
  226.     //  Did we connect?
  227.     if ($this->pop_conn == false) {
  228.       //  It would appear not...
  229.       $this->error = array(
  230.         'error' => "Failed to connect to server $host on port $port",
  231.         'errno' => $errno,
  232.         'errstr' => $errstr
  233.       );
  234.  
  235.       if ($this->do_debug >= 1) {
  236.         $this->displayErrors();
  237.       }
  238.  
  239.       return false;
  240.     }
  241.  
  242.     //  Increase the stream time-out
  243.  
  244.     //  Check for PHP 4.3.0 or later
  245.     if (version_compare(phpversion(), '4.3.0', 'ge')) {
  246.       stream_set_timeout($this->pop_conn, $tval, 0);
  247.     } else {
  248.       //  Does not work on Windows
  249.       if (substr(PHP_OS, 0, 3) !== 'WIN') {
  250.         socket_set_timeout($this->pop_conn, $tval, 0);
  251.       }
  252.     }
  253.  
  254.     //  Get the POP3 server response
  255.     $pop3_response = $this->getResponse();
  256.  
  257.     //  Check for the +OK
  258.     if ($this->checkResponse($pop3_response)) {
  259.     //  The connection is established and the POP3 server is talking
  260.     $this->connected = true;
  261.       return true;
  262.     }
  263.  
  264.   }
  265.  
  266.   /**
  267.    * Login to the POP3 server (does not support APOP yet)
  268.    *
  269.    * @param string $username 
  270.    * @param string $password 
  271.    * @return boolean 
  272.    */
  273.   function Login ($username = '', $password = '') {
  274.     if ($this->connected == false) {
  275.       $this->error = 'Not connected to POP3 server';
  276.  
  277.       if ($this->do_debug >= 1) {
  278.         $this->displayErrors();
  279.       }
  280.     }
  281.  
  282.     if (empty($username)) {
  283.       $username = $this->username;
  284.     }
  285.  
  286.     if (empty($password)) {
  287.       $password = $this->password;
  288.     }
  289.  
  290.     $pop_username = "USER $username" . $this->CRLF;
  291.     $pop_password = "PASS $password" . $this->CRLF;
  292.  
  293.     //  Send the Username
  294.     $this->sendString($pop_username);
  295.     $pop3_response = $this->getResponse();
  296.  
  297.     if ($this->checkResponse($pop3_response)) {
  298.       //  Send the Password
  299.       $this->sendString($pop_password);
  300.       $pop3_response = $this->getResponse();
  301.  
  302.       if ($this->checkResponse($pop3_response)) {
  303.         return true;
  304.       } else {
  305.         return false;
  306.       }
  307.     } else {
  308.       return false;
  309.     }
  310.   }

  313.    * Disconnect from the POP3 server

  316.     $this->sendString('QUIT');
  317.  
  318.     fclose($this->pop_conn);
  319.   }

  358.       $this->error = array(
  359.         'error' => "Server reported an error: $string",
  360.         'errno' => 0,

  362.       );
  363.  
  364.       if ($this->do_debug >= 1) {
  365.         $this->displayErrors();
  366.       }
  367.  
  368.       return false;
  369.     } else {
  370.       return true;
  371.     }
  372.  
  373.   }
